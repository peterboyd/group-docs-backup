Working samba config

Note: This is SAMBA config I used on one server. I am posting it here because it works for sure, so you can test your own servers when in doubt.
Note: I do not wouch that these settings are good, secure, etc., all I am saying is that it works in protected network.
Note: You can modify only most necessary items and test if it works for you. When you see it works, then change one setting at the time, restart samba, check, and if it works change another setting.

[global]
    log file = /var/log/samba/log.%m
    netbios name = server
    guest account = nobody
    netbios aliases = linux
    writeable = yes
    server string = Server
    log level = audit:3
    sync always = yes
    workgroup = VODOREMONT
    guest ok = yes
    os level = 20
    auto services = global homes printers
    username map = /etc/samba/smbusers
    encrypt passwords = yes
    security = share
    passdb backend = tdbsam
    max log size = 50
    wins support = true
;    allow hosts = 192.168.1.0/24
;    cups options = raw
;    load printers = yes
;    restrict anonymous = 0
;    null passwords = no
;       ----------
;       -----------
;    username map = /etc/samba/smbusers


[homes]
    comment = Home Directories
    browseable = yes
    writable = yes
;    valid users = %S
;    valid users = MYDOMAIN\%S
   
[printers]
    comment = All Printers
    path = /var/spool/samba
    browseable = yes
    guest ok = yes
    writable = no
    printable = yes
   
[d-disk]
    browseable = yes
    guest ok = yes
;    guest account = tanja
    writeable = yes
    delete readonly = yes
    public = yes
    path = /extra
    sync always = yes
    read only = no
