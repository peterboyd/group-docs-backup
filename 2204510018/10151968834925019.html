Script to find IPv6 addresses for all network devices that have been assigned an IPv4 address.

#!/bin/sh

LIST="`ip -4 addr show | grep "inet.*global" | awk '{print $NF}'`"

#!/bin/bash

for line in $(ifconfig | grep '^[a-z0-9]' | awk '{print $1}'); do

if [[ "$(ip addr show dev $line | sed -e's/^.*inet6 \([^ ]*\)\/.*$/\1/;t;d')" != "::1" && "$(ip addr show dev $line | sed -e's/^.*inet6 \([^ ]*\)\/.*$/\1/;t;d')" != "" && "$(ip addr show dev $line | grep -c 'inet ')" == "1" ]]; then

echo -ne $line; echo -ne ' '

ip addr show dev $line | sed -e's/^.*inet6 \([^ ]*\)\/.*$/\1/;t;d'

fi

done
